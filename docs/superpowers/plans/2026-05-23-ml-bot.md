**Goal:** Remplacer le bot heuristique de COMBO par un agent PPO entraîné via Reinforcement Learning, avec trois niveaux de difficulté (easy / medium / hard), servi via un microservice FastAPI sur le VPS.

**Architecture:** Un environnement gymnasium Python simule le jeu COMBO. Trois modèles PPO (MaskablePPO de sb3-contrib) sont entraînés en self-play contre un adversaire heuristique. Un microservice FastAPI charge ces modèles et répond aux requêtes de Node.js. Le serveur Node.js appelle ce microservice à chaque tour de bot avec un fallback sur les heuristiques actuelles si le service est indisponible.

**Tech Stack:** Python 3.11+, gymnasium, stable-baselines3, sb3-contrib (MaskablePPO), FastAPI, uvicorn, Node.js fetch (natif Node 22+), systemd.

```python
# ml/cards.py
import random
from typing import Optional

# suit: 'H'=Coeurs, 'D'=Carreaux, 'C'=Trèfles, 'S'=Piques
# rank: 0=As, 1=2, 2=3, ..., 9=10, 10=Valet, 11=Dame, 12=Roi
SUITS = ['H', 'D', 'C', 'S']
RANKS = list(range(13))


def card_value(rank: int) -> int:
    """Points de la carte (As=0, 2-10=valeur faciale, V=11, D=12, R=13)."""
    if rank == 0:
        return 0
    return rank + 1


def is_red_head(rank: int, suit: str) -> bool:
    """Vrai si la carte est une tête rouge (V/D/R de Coeurs ou Carreaux)."""
    return rank >= 10 and suit in ('H', 'D')


def power_type(rank: int) -> Optional[str]:
    """Retourne le type de pouvoir d'une tête rouge, ou None."""
    if rank == 10:
        return 'self-peek'
    if rank == 11:
        return 'opp-peek'
    if rank == 12:
        return 'swap'
    return None


def make_deck() -> list[tuple[int, str]]:
    return [(rank, suit) for suit in SUITS for rank in RANKS]


def shuffled_deck() -> list[tuple[int, str]]:
    deck = make_deck()
    random.shuffle(deck)
    return deck
```

## Task 3 : Environnement gymnasium — structure et observation (ml/combo_env.py)

**Files:**
- Create: `ml/combo_env.py`

L'environnement simule un duel bot vs 1 adversaire heuristique.

**Phases (encodées 0–3) :**
- `0` PHASE_TURN_START : bot choisit draw_deck / draw_discard / call_combo
- `1` PHASE_AFTER_DRAW : bot choisit swap_0..3 / discard
- `2` PHASE_SNAP : bot choisit snap_0..3 / pass_snap (après discard de l'adversaire)
- `3` PHASE_POWER : bot résout un pouvoir tête rouge

**Observation (vecteur 15 float32) :**
```
[hand_0..3, discard_top, drawn_card, bot_total, opp_total, 0, 0, opp_count, 0, 0, round_num, phase]
```
- `hand_i` : valeur de la carte si connue, -1.0 si inconnue ou trou
- `discard_top` : valeur du dessus de la défausse (0 si vide)
- `drawn_card` : valeur de la carte piochée (-1 si pas encore pioché)
- `bot_total / opp_total` : scores cumulés sur les manches précédentes
- `opp_count` : nombre de cartes restantes de l'adversaire
- `round_num` : numéro de manche courante (1–N)
- `phase` : 0–3

**Actions (25 Discrete) :**
| Index | Action | Phase |
|---|---|---|
| 0 | draw_deck | 0 |
| 1 | draw_discard | 0 |
| 2 | call_combo | 0 |
| 3–6 | swap_slot_0..3 | 1 |
| 7 | discard | 1 |
| 8–11 | snap_0..3 | 2 |
| 12 | pass_snap | 2 |
| 13–16 | peek_own_0..3 | 3 (self-peek) |
| 17–20 | peek_opp_0..3 | 3 (opp-peek) |
| 21–24 | swap_opp_0..3 | 3 (swap — bot échange sa pire carte connue avec opp slot X) |

```python
# ml/combo_env.py
import gymnasium as gym
import numpy as np
from gymnasium import spaces
from cards import card_value, is_red_head, power_type, shuffled_deck

PHASE_TURN_START = 0
PHASE_AFTER_DRAW = 1
PHASE_SNAP = 2
PHASE_POWER = 3

N_ACTIONS = 25
N_OBS = 15


class ComboEnv(gym.Env):
    """Environnement COMBO : bot (agent) vs 1 adversaire heuristique."""

    metadata = {"render_modes": []}

    def __init__(self, rounds: int = 3):
        super().__init__()
        self.rounds = rounds
        self.observation_space = spaces.Box(
            low=-1.0, high=200.0, shape=(N_OBS,), dtype=np.float32
        )
        self.action_space = spaces.Discrete(N_ACTIONS)

        # État interne (initialisé dans reset)
        self.bot_hand: list = []
        self.opp_hand: list = []
        self.bot_known: list[bool] = []
        self.deck: list = []
        self.discard: list = []
        self.drawn_card = None
        self.drawn_from_discard: bool = False
        self.phase: int = PHASE_TURN_START
        self.bot_total: int = 0
        self.opp_total: int = 0
        self.round_num: int = 1
        self._pending_power: str | None = None
        self._combo_called: bool = False
        self._combo_by_bot: bool = False

    # ------------------------------------------------------------------
    # gymnasium API
    # ------------------------------------------------------------------

    def reset(self, seed=None, options=None):
        super().reset(seed=seed)
        self._init_round()
        self.bot_total = 0
        self.opp_total = 0
        self.round_num = 1
        return self._obs(), {}

    def step(self, action: int):
        reward = 0.0
        terminated = False

        if self.phase == PHASE_TURN_START:
            reward, terminated = self._handle_turn_start(action)
        elif self.phase == PHASE_AFTER_DRAW:
            reward, terminated = self._handle_after_draw(action)
        elif self.phase == PHASE_SNAP:
            reward, terminated = self._handle_snap(action)
        elif self.phase == PHASE_POWER:
            reward, terminated = self._handle_power(action)

        return self._obs(), reward, terminated, False, {}

    def action_masks(self) -> np.ndarray:
        mask = np.zeros(N_ACTIONS, dtype=bool)
        if self.phase == PHASE_TURN_START:
            mask[0] = bool(self.deck)       # draw_deck
            mask[1] = bool(self.discard)    # draw_discard
            if self._estimate_bot_score() < 8:
                mask[2] = True              # call_combo
            if not any(mask):
                mask[0] = True              # garantit au moins une action
        elif self.phase == PHASE_AFTER_DRAW:
            for i in range(4):
                mask[3 + i] = True          # swap_slot_i toujours possible
            if not self.drawn_from_discard:
                mask[7] = True              # discard (sauf si pioché depuis discard)
            if not any(mask):
                mask[7] = True
        elif self.phase == PHASE_SNAP:
            if self.discard:
                top_rank = self.discard[-1][0]
                for i in range(4):
                    card = self.bot_hand[i]
                    if card is not None and card[0] == top_rank:
                        mask[8 + i] = True
            mask[12] = True                 # pass_snap toujours valide
        elif self.phase == PHASE_POWER:
            p = self._pending_power
            if p == 'self-peek':
                for i in range(4):
                    if self.bot_hand[i] is not None:
                        mask[13 + i] = True
            elif p == 'opp-peek':
                for i in range(4):
                    if self.opp_hand[i] is not None:
                        mask[17 + i] = True
            elif p == 'swap':
                for i in range(4):
                    if self.opp_hand[i] is not None:
                        mask[21 + i] = True
            if not any(mask):
                mask[12] = True             # fallback pass
        return mask

    # ------------------------------------------------------------------
    # Handlers de phase
    # ------------------------------------------------------------------

    def _handle_turn_start(self, action: int) -> tuple[float, bool]:
        if action == 0:  # draw_deck
            self.drawn_card = self.deck.pop() if self.deck else None
            self.drawn_from_discard = False
            self.phase = PHASE_AFTER_DRAW
            return 0.0, False
        if action == 1:  # draw_discard
            self.drawn_card = self.discard.pop() if self.discard else None
            self.drawn_from_discard = True
            self.phase = PHASE_AFTER_DRAW
            return 0.0, False
        if action == 2:  # call_combo
            self._combo_called = True
            self._combo_by_bot = True
            # Adversaire joue son dernier tour
            self._play_opponent_full_turn()
            return self._end_round()
        return 0.0, False

    def _handle_after_draw(self, action: int) -> tuple[float, bool]:
        if 3 <= action <= 6:  # swap_slot_i
            slot = action - 3
            old_card = self.bot_hand[slot]
            self.bot_hand[slot] = self.drawn_card
            self.bot_known[slot] = True
            if old_card is not None:
                self.discard.append(old_card)
            self.drawn_card = None
            self.drawn_from_discard = False
            self.phase = PHASE_TURN_START
            # Adversaire joue → bot aura une window de snap
            self._play_opponent_full_turn_with_snap()
            return 0.0, False

        if action == 7:  # discard
            card = self.drawn_card
            self.discard.append(card)
            self.drawn_card = None
            self.drawn_from_discard = False
            if is_red_head(card[0], card[1]):
                self._pending_power = power_type(card[0])
                self.phase = PHASE_POWER
            else:
                self.phase = PHASE_TURN_START
                self._play_opponent_full_turn_with_snap()
            return 0.0, False
        return 0.0, False

    def _handle_snap(self, action: int) -> tuple[float, bool]:
        reward = 0.0
        if 8 <= action <= 11:  # snap_i
            slot = action - 8
            card = self.bot_hand[slot]
            if card is not None and self.discard and card[0] == self.discard[-1][0]:
                self.bot_hand[slot] = None
                self.bot_known[slot] = False
                reward += 0.3
                if all(c is None for c in self.bot_hand):
                    reward += 2.0  # perfect snap
                    r, terminated = self._end_round()
                    self.phase = PHASE_TURN_START
                    return reward + r, terminated
            else:
                # Snap raté : carte penalty
                if self.deck:
                    self.bot_hand.append(self.deck.pop())
                    self.bot_known.append(False)
                reward -= 0.2
        # pass_snap (12) : rien
        self.phase = PHASE_TURN_START
        return reward, False

    def _handle_power(self, action: int) -> tuple[float, bool]:
        p = self._pending_power
        if p == 'self-peek' and 13 <= action <= 16:
            slot = action - 13
            if self.bot_hand[slot] is not None:
                self.bot_known[slot] = True
        elif p == 'opp-peek' and 17 <= action <= 20:
            pass  # bot voit une carte adverse (pas d'effet sur notre obs simplifiée)
        elif p == 'swap' and 21 <= action <= 24:
            opp_slot = action - 21
            own_slot = self._bot_worst_slot()
            if own_slot is not None and self.opp_hand[opp_slot] is not None:
                self.bot_hand[own_slot], self.opp_hand[opp_slot] = (
                    self.opp_hand[opp_slot],
                    self.bot_hand[own_slot],
                )
                self.bot_known[own_slot] = False  # carte inconnue reçue

        self._pending_power = None
        self.phase = PHASE_TURN_START
        self._play_opponent_full_turn_with_snap()
        return 0.0, False

    # ------------------------------------------------------------------
    # Adversaire heuristique
    # ------------------------------------------------------------------

    def _play_opponent_full_turn_with_snap(self):
        """Adversaire joue son tour complet, puis la phase snap est offerte au bot."""
        self._play_opponent_full_turn()
        # Après le tour adverse, le bot obtient une window de snap (phase 2)
        self.phase = PHASE_SNAP

    def _play_opponent_full_turn(self):
        """Heuristique simple : pioche, échange si la carte améliore sa main, sinon défausse."""
        if not self.deck:
            return
        card = self.deck.pop()
        card_val = card_value(card[0])

        known_vals = [card_value(c[0]) for c in self.opp_hand if c is not None]
        avg = sum(known_vals) / len(known_vals) if known_vals else 6.5

        if card_val < avg:
            # Échange avec la pire carte
            worst = max(
                (i for i in range(len(self.opp_hand)) if self.opp_hand[i] is not None),
                key=lambda i: card_value(self.opp_hand[i][0]),
                default=None,
            )
            if worst is not None:
                old = self.opp_hand[worst]
                self.opp_hand[worst] = card
                self.discard.append(old)
                return
        # Défausse
        self.discard.append(card)
        # Pouvoir tête rouge adverse : auto-résolution simplifiée (pas d'effet)

    # ------------------------------------------------------------------
    # Fin de manche / partie
    # ------------------------------------------------------------------

    def _end_round(self) -> tuple[float, bool]:
        bot_score = sum(
            card_value(c[0]) for c in self.bot_hand if c is not None
        )
        opp_score = sum(
            card_value(c[0]) for c in self.opp_hand if c is not None
        )

        # Pénalité combo
        if self._combo_called:
            if self._combo_by_bot and bot_score >= opp_score:
                bot_score += 10   # bot a appelé mais n'avait pas le min strict
            elif not self._combo_by_bot and opp_score >= bot_score:
                opp_score += 10

        # Récompense de manche
        reward = 0.0
        if bot_score < opp_score:
            reward += 1.0
        elif bot_score > opp_score:
            reward -= 0.5
        reward += (opp_score - bot_score) / 50.0

        self.bot_total += bot_score
        self.opp_total += opp_score
        self._combo_called = False
        self._combo_by_bot = False

        if self.round_num >= self.rounds:
            # Fin de partie
            if self.bot_total < self.opp_total:
                reward += 2.0
            elif self.bot_total > self.opp_total:
                reward -= 1.0
            return reward, True

        # Nouvelle manche
        self.round_num += 1
        self._init_round()
        return reward, False

    # ------------------------------------------------------------------
    # Helpers
    # ------------------------------------------------------------------

    def _init_round(self):
        deck = shuffled_deck()
        self.bot_hand = [deck.pop() for _ in range(4)]
        self.opp_hand = [deck.pop() for _ in range(4)]
        self.deck = deck
        self.discard = [self.deck.pop()]
        self.bot_known = [False, False, True, True]  # mémorise les 2 dernières
        self.drawn_card = None
        self.drawn_from_discard = False
        self.phase = PHASE_TURN_START
        self._pending_power = None

    def _obs(self) -> np.ndarray:
        hand_vals = []
        for i, card in enumerate(self.bot_hand[:4]):
            if card is None:
                hand_vals.append(-1.0)
            elif self.bot_known[i]:
                hand_vals.append(float(card_value(card[0])))
            else:
                hand_vals.append(-1.0)
        while len(hand_vals) < 4:
            hand_vals.append(-1.0)

        discard_top = float(card_value(self.discard[-1][0])) if self.discard else 0.0
        drawn = float(card_value(self.drawn_card[0])) if self.drawn_card else -1.0
        opp_count = float(sum(1 for c in self.opp_hand if c is not None))

        return np.array([
            *hand_vals,               # 0–3  : main du bot
            discard_top,              # 4    : dessus discard
            drawn,                    # 5    : carte piochée
            float(self.bot_total),    # 6    : score cumulé bot
            float(self.opp_total),    # 7    : score cumulé adv
            0.0, 0.0,                 # 8–9  : slots inutilisés (multi-adv)
            opp_count,                # 10   : nb cartes adv
            0.0, 0.0,                 # 11–12: slots inutilisés
            float(self.round_num),    # 13   : numéro de manche
            float(self.phase),        # 14   : phase
        ], dtype=np.float32)

    def _estimate_bot_score(self) -> float:
        total = 0.0
        for i, card in enumerate(self.bot_hand):
            if card is None:
                continue
            total += card_value(card[0]) if self.bot_known[i] else 6.5
        return total

    def _bot_worst_slot(self) -> int | None:
        worst_val, worst_slot = -1, None
        for i, card in enumerate(self.bot_hand):
            if card is not None and self.bot_known[i]:
                v = card_value(card[0])
                if v > worst_val:
                    worst_val, worst_slot = v, i
        return worst_slot
```

```python
# ml/server.py
import os
import numpy as np
import random
from contextlib import asynccontextmanager
from fastapi import FastAPI
from pydantic import BaseModel

MODELS_DIR = os.path.join(os.path.dirname(__file__), "models")
MODELS: dict = {}

ACTION_NAMES = [
    "draw_deck", "draw_discard", "call_combo",
    "swap_0", "swap_1", "swap_2", "swap_3", "discard",
    "snap_0", "snap_1", "snap_2", "snap_3", "pass_snap",
    "peek_own_0", "peek_own_1", "peek_own_2", "peek_own_3",
    "peek_opp_0", "peek_opp_1", "peek_opp_2", "peek_opp_3",
    "swap_opp_0", "swap_opp_1", "swap_opp_2", "swap_opp_3",
]


def load_models():
    try:
        from sb3_contrib import MaskablePPO
        for level in ("easy", "medium", "hard"):
            path = os.path.join(MODELS_DIR, f"model_{level}.zip")
            if os.path.exists(path):
                MODELS[level] = MaskablePPO.load(path)
                print(f"[combo-ai] Modèle '{level}' chargé.")
            else:
                print(f"[combo-ai] Modèle '{level}' introuvable ({path}), ignoré.")
    except Exception as e:
        print(f"[combo-ai] Erreur chargement modèles : {e}")


@asynccontextmanager
async def lifespan(app: FastAPI):
    load_models()
    yield


app = FastAPI(lifespan=lifespan)


class GameState(BaseModel):
    hand: list[float]           # 4 valeurs
    discard_top: float
    drawn_card: float
    bot_total: float
    opp_total: float
    opp_count: float
    round_num: float
    phase: float


class ActionRequest(BaseModel):
    state: GameState
    difficulty: str             # "easy" | "medium" | "hard"
    legal_actions: list[int]    # indices des actions légales


class ActionResponse(BaseModel):
    action_index: int
    action_name: str


def predict_action(model, obs: np.ndarray, legal_actions: list[int]) -> int:
    mask = np.zeros(25, dtype=bool)
    for i in legal_actions:
        mask[i] = True
    action, _ = model.predict(obs, action_masks=mask, deterministic=True)
    return int(action)


def state_to_obs(state: GameState) -> np.ndarray:
    hand = list(state.hand[:4])
    while len(hand) < 4:
        hand.append(-1.0)
    return np.array([
        *hand,
        state.discard_top,
        state.drawn_card,
        state.bot_total,
        state.opp_total,
        0.0, 0.0,
        state.opp_count,
        0.0, 0.0,
        state.round_num,
        state.phase,
    ], dtype=np.float32)


@app.get("/health")
def health():
    return {"status": "ok", "models_loaded": list(MODELS.keys())}


@app.post("/action", response_model=ActionResponse)
def action(req: ActionRequest):
    legal = req.legal_actions
    if not legal:
        legal = [0]

    model = MODELS.get(req.difficulty)
    if model is None:
        # Fallback : action légale aléatoire
        idx = random.choice(legal)
    else:
        obs = state_to_obs(req.state)
        idx = predict_action(model, obs, legal)

    # Sécurité : si le modèle retourne une action illégale, prendre la première légale
    if idx not in legal:
        idx = legal[0]

    return ActionResponse(action_index=idx, action_name=ACTION_NAMES[idx])
```

```typescript
export async function askMLBot(
  state: MLBotState,
  difficulty: BotDifficulty,
  legalActions: number[],
): Promise<MLAction | null> {
  try {
    const res = await fetch(`${ML_API}/action`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ state, difficulty, legal_actions: legalActions }),
      signal: AbortSignal.timeout(500),
    });

  } catch {
    return null;  // timeout ou service down → fallback heuristiques
  }
}

export function cardValue(rank: number): number {
```
